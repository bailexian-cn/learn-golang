# 打印结构体的内存地址

```golang
package main

import "fmt"

func main() {
	p1 := People{
		Name: "Lucy",
		Age: 32,
	}
	p1Copy := p1
	fmt.Println(fmt.Sprintf("p1 memory address is %p", &p1))
	fmt.Println(fmt.Sprintf("p1Copy memory address is %p", &p1Copy))
}

type People struct {
	Name string `json:"name,omitempty"`
	Age  uint8  `json:"age,omitempty"`
}
```
